üìã Stripe Configuration Script
Use this in your setup:
typescript// /scripts/setup-stripe-products.ts
import Stripe from 'stripe';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: '2024-11-20.acacia',
});

async function setupStripeProducts() {
  // 1. STARTER PLAN
  const starterProduct = await stripe.products.create({
    name: 'MarketPulse - Starter Plan',
    description: 'Monitor up to 5 local competitors with daily price tracking and instant email alerts.',
    metadata: {
      plan_name: 'starter',
      competitor_limit: '5',
      features: 'price_tracking,email_alerts,mobile_access',
      crawl_frequency: 'daily',
      history_days: '30',
      support_level: 'email',
      sms_alerts: 'false',
      api_access: 'false',
    },
  });

  const starterPrice = await stripe.prices.create({
    product: starterProduct.id,
    unit_amount: 4900, // $49.00
    currency: 'usd',
    recurring: {
      interval: 'month',
      trial_period_days: 14,
    },
    nickname: 'Starter - Monthly',
  });

  console.log('‚úÖ Starter Plan Created');
  console.log('   Product ID:', starterProduct.id);
  console.log('   Price ID:', starterPrice.id);

  // 2. PROFESSIONAL PLAN
  const proProduct = await stripe.products.create({
    name: 'MarketPulse - Professional Plan',
    description: 'Monitor up to 20 competitors with twice-daily tracking, SMS alerts, and webhook integrations.',
    metadata: {
      plan_name: 'professional',
      competitor_limit: '20',
      features: 'price_tracking,email_alerts,sms_alerts,webhooks,analytics,mobile_access,csv_export',
      crawl_frequency: 'twice_daily',
      history_days: '90',
      support_level: 'priority_email',
      sms_alerts: 'true',
      sms_limit_monthly: '100',
      api_access: 'false',
      webhook_destinations: '5',
    },
  });

  const proMonthlyPrice = await stripe.prices.create({
    product: proProduct.id,
    unit_amount: 9900, // $99.00
    currency: 'usd',
    recurring: {
      interval: 'month',
      trial_period_days: 14,
    },
    nickname: 'Professional - Monthly',
  });

  const proAnnualPrice = await stripe.prices.create({
    product: proProduct.id,
    unit_amount: 94800, // $948.00 ($79/month)
    currency: 'usd',
    recurring: {
      interval: 'year',
      trial_period_days: 14,
    },
    nickname: 'Professional - Annual (Save $240)',
  });

  console.log('‚úÖ Professional Plan Created');
  console.log('   Product ID:', proProduct.id);
  console.log('   Monthly Price ID:', proMonthlyPrice.id);
  console.log('   Annual Price ID:', proAnnualPrice.id);

  // 3. ENTERPRISE PLAN
  const enterpriseProduct = await stripe.products.create({
    name: 'MarketPulse - Enterprise Plan',
    description: 'Unlimited competitor monitoring with hourly tracking, API access, and dedicated support.',
    metadata: {
      plan_name: 'enterprise',
      competitor_limit: 'unlimited',
      features: 'price_tracking,email_alerts,sms_alerts,webhooks,analytics,mobile_access,csv_export,api_access,multi_location,ai_insights,custom_rules',
      crawl_frequency: 'hourly',
      history_days: 'unlimited',
      support_level: 'dedicated_phone_email',
      sms_alerts: 'true',
      sms_limit_monthly: 'unlimited',
      api_access: 'true',
      webhook_destinations: 'unlimited',
      account_manager: 'true',
    },
  });

  const enterpriseMonthlyPrice = await stripe.prices.create({
    product: enterpriseProduct.id,
    unit_amount: 29900, // $299.00
    currency: 'usd',
    recurring: {
      interval: 'month',
      trial_period_days: 14,
    },
    nickname: 'Enterprise - Monthly',
  });

  const enterpriseAnnualPrice = await stripe.prices.create({
    product: enterpriseProduct.id,
    unit_amount: 298800, // $2,988.00 ($249/month)
    currency: 'usd',
    recurring: {
      interval: 'year',
      trial_period_days: 14,
    },
    nickname: 'Enterprise - Annual (Save $600)',
  });

  console.log('‚úÖ Enterprise Plan Created');
  console.log('   Product ID:', enterpriseProduct.id);
  console.log('   Monthly Price ID:', enterpriseMonthlyPrice.id);
  console.log('   Annual Price ID:', enterpriseAnnualPrice.id);

  console.log('\nüéâ All Stripe products created successfully!');
  console.log('\nAdd these to your .env:');
  console.log(`STRIPE_STARTER_PRICE_ID=${starterPrice.id}`);
  console.log(`STRIPE_PRO_MONTHLY_PRICE_ID=${proMonthlyPrice.id}`);
  console.log(`STRIPE_PRO_ANNUAL_PRICE_ID=${proAnnualPrice.id}`);
  console.log(`STRIPE_ENTERPRISE_MONTHLY_PRICE_ID=${enterpriseMonthlyPrice.id}`);
  console.log(`STRIPE_ENTERPRISE_ANNUAL_PRICE_ID=${enterpriseAnnualPrice.id}`);
}

setupStripeProducts().catch(console.error);
